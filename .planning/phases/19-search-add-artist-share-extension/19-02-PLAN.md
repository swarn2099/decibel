**Final approach for useAddArtist**:
```typescript
export function useAddArtist() {
  return useMutation<AddArtistResult, Error, { spotifyId: string, name: string, photoUrl: string | null, genres: string[], followers: number, monthlyListeners?: number }>({
    mutationFn: async ({ spotifyId, name, photoUrl, genres, followers, monthlyListeners }) => {
      // 1. Auth
      const { data: { user } } = await supabase.auth.getUser();
      if (!user?.email) throw new Error("Not authenticated");

      const { data: fan } = await supabase.from("fans").select("id").eq("email", user.email).single();
      if (!fan) throw new Error("Fan profile not found");

      // 2. Check if performer exists
      const { data: existing } = await supabase
        .from("performers")
        .select("id, name, slug")
        .eq("spotify_id", spotifyId)
        .maybeSingle();

      if (existing) {
        const { data: founder } = await supabase
          .from("founder_badges")
          .select("fan_id, fans(display_name, email)")
          .eq("performer_id", existing.id)
          .maybeSingle();
        return { already_exists: true, performer: existing, founder_name: founder?.fans?.display_name || founder?.fans?.email || "Someone", is_founder: false };
      }

      // 3. Create performer
      const slug = generateSlug(name); // same slug logic as web
      const { data: performer, error } = await supabase
        .from("performers")
        .insert({
          name, slug, photo_url: photoUrl,
          genres, follower_count: followers,
          spotify_id: spotifyId,
          spotify_url: `https://open.spotify.com/artist/${spotifyId}`,
          monthly_listeners: monthlyListeners,
          claimed: false,
        })
        .select("id, name, slug")
        .single();

      if (error) throw new Error("Failed to create performer");

      // 4. Auto-discover (add to collection)
      await supabase.from("collections").insert({
        fan_id: fan.id, performer_id: performer.id,
        capture_method: "online", verified: false,
      });

      // 5. Founder badge (under 1M only)
      let isFounder = false;
      if (followers < 1_000_000) {
        const { error: founderErr } = await supabase
          .from("founder_badges")
          .insert({ fan_id: fan.id, performer_id: performer.id });
        isFounder = !founderErr;
      }

      // 6. Upsert fan_tiers
      await supabase.from("fan_tiers").upsert({
        fan_id: fan.id, performer_id: performer.id,
        scan_count: 1, current_tier: "network",
      }, { onConflict: "fan_id,performer_id" });

      return { already_exists: false, performer, is_founder: isFounder, founder_name: null };
    },
    onSuccess: (result) => {
      if (result.is_founder) {
        Haptics.impactAsync(Haptics.ImpactFeedbackStyle.Heavy);
      } else {
        Haptics.impactAsync(Haptics.ImpactFeedbackStyle.Medium);
      }
    },
  });
}
```
