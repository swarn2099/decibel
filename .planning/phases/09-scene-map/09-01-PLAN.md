export interface MapEvent {

  performer_name: string;

  id: string;
  name: string;
  slug: string;
  latitude: number;
  longitude: number;
  event_count: number;
  genres: string[];

}

   - Logs progress: "Geocoded {name}: {lat}, {lng}" or "Could not geocode {name}, using jittered default"

<task type="auto">
  <name>Task 2: Create /api/map endpoint</name>
  <files>src/app/api/map/route.ts</files>
  <action>
Create GET endpoint at `/api/map` that:

1. Queries Supabase for all venues that have at least one upcoming event (event_date >= today)
2. For each venue, fetches upcoming events with performer info via join:
   ```
   supabase.from("venues").select(`
     id, name, slug, latitude, longitude,
     events!inner(
       id, event_date, start_time, external_url,
       performer:performers(name, slug, photo_url, genres)
     )
   `).gte("events.event_date", today)
   ```
3. Transforms the response into MapVenue[] format:
   - `event_count`: number of upcoming events at this venue
   - `genres`: deduplicated flat array of all genres from all performers playing there (from performers.genres which is a text[] column)
   - `upcoming_events`: array of MapEvent objects sorted by event_date ascending
4. Returns JSON with `{ venues: MapVenue[] }`
5. Add `export const revalidate = 300` for 5-minute ISR caching (matches homepage pattern)

Use `createSupabaseServer()` from `@/lib/supabase-server`.

Accept optional query params:
- `genre` — filter venues to only those hosting events with performers matching this genre
- `tonight` — if "true", filter events to today's date only

Apply filters at the Supabase query level for efficiency.
  </action>
